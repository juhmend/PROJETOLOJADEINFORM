/*CRIANDO BANCO DE DADOS*/

CREATE DATABASE PROJETOLOJADEINFORM

/*USANDO BANCO DE DADOS*/

USE PROJETOLOJADEINFORM

/*CRIANDO TABELAS DO BANCO DE DADOS*/

CREATE TABLE FUNCIONARIO(
COD_FUNC INT PRIMARY KEY IDENTITY,
NOME_FUNC VARCHAR(50),
TELEFONE_FUNC VARCHAR(15),
EMAIL_FUNC VARCHAR(50)
)

CREATE TABLE PRODUTO(
COD_PROD INT PRIMARY KEY IDENTITY,
NOME_PROD VARCHAR(50),
DESCRICAO_PROD VARCHAR(50),
QUANT_PROD INT
)

CREATE TABLE CLIENTE(
COD_CLI INT PRIMARY KEY IDENTITY,
NOME_CLI VARCHAR(50),
TELEFONE_CLI VARCHAR(15),
EMAIL_CLI VARCHAR(50)
)

CREATE TABLE PEDIDO(
COD_PED INT PRIMARY KEY IDENTITY,
FK_COD_CLI INT,
FK_COD_PROD INT,
DESCRICAO_PED VARCHAR(50),
VALOR_PED DOUBLE,
FOREIGN KEY (FK_COD_CLI) REFERENCES CLIENTE (COD_CLI),
FOREIGN KEY (FK_COD_PROD) REFERENCES PRODUTO (COD_PROD)
)

CREATE TABLE VENDA(
COD_VENDA INT PRIMARY KEY IDENTITY,
FK_COD_PED INT,
FOREIGN KEY (FK_COD_PED) REFERENCES PEDIDO (COD_PED)
)

/*CONSULTANDO AS TABELAS DE BANCO DE DADOS*/

SELECT * FROM FUNCIONARIO
SELECT * FROM PRODUTO
SELECT * FROM CLIENTE
SELECT * FROM PEDIDO
SELECT * FROM VENDA
